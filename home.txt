// Tasks 8 Napishite programmu, kotoraya na vkhod prinimayet chislo (N), 
//a na vykhode pokazyvayet vse chotnyye chisla ot 1 do N.

Console.Write("Vvedite naturalnoe chislo: ");
int N = int.Parse(Console.ReadLine());

int count = 0;

if (N < 0)
{
    while (count > N+1)
   {
    count -= 2;    
    Console.Write($" {count}");
   }
}
else
{
    while (count < N-1)
   {
    count += 2;    
    Console.Write($" {count}");
   }
}
